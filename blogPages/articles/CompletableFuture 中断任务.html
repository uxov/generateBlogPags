CompletableFuture - Interrupt A Task

future.cancel(true) can't interrupt a task

public class Test {
    public static void main(String[] args) {
        CompletableFuture future = null;
        try {
            future = CompletableFuture.runAsync(() -> {
                int count = 0;
                while (true) {
                    System.out.println("run async - " + count);
                    count++;
                }
            });
            future.get(1, TimeUnit.SECONDS);
        } catch (TimeoutException te) {     //the code in loop continue running when timeout
            te.printStackTrace();
            future.cancel(true);    //doesn't work
        } catch (Exception e) {
            e.printStackTrace();
        }
        try {
            Thread.currentThread().join();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
}

The right way to interrupt a task:

public class Test {
    public static void main(String[] args) {
        CompletableFuture future = null;
        AtomicBoolean wait = new AtomicBoolean(true);
        try {
            future = CompletableFuture.runAsync(() -> {
                int count = 0;
                while (wait.get()) {
                    System.out.println("run async - " + count);
                    count++;
                }
            });
            future.get(1, TimeUnit.SECONDS);
        } catch (TimeoutException te) {
            te.printStackTrace();
            wait.set(false);    //terminate the loop
        } catch (Exception e) {
            e.printStackTrace();
        }
        try {
            Thread.currentThread().join();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
}
